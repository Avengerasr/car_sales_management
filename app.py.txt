from flask import Flask, render_template, request
import pymysql

app = Flask(__name__)

def get_db_connection():
    return pymysql.connect(
        host='localhost',
        user='root',
        password='',
        database='car_sales_db',
        cursorclass=pymysql.cursors.DictCursor
    )

@app.route('/')
def index():
    conn = get_db_connection()
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM Car WHERE Status='Available'")
    cars = cursor.fetchall()
    conn.close()
    return render_template('index.html', cars=cars)

@app.route('/sell/<int:car_id>')
def sell_car(car_id):
    conn = get_db_connection()
    cursor = conn.cursor()
    cursor.execute("UPDATE Car SET Status='Sold' WHERE Car_ID=%s", (car_id,))
    conn.commit()
    conn.close()
    return "Car sold successfully!"

if __name__ == '__main__':
    app.run(debug=True)
